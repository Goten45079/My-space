sample ansible-playbook

---
- name: Name of the play book
  hosts: all or any_group_name
  become: true # this means you want to execute the commands as root user
  gather_facts: yes

  tasks:
  - name: Name of the task. example install nginx
    apt:
     name: nginx
     state: present

or you have to install nginx on different linux 

  tasks:
  - name: install nginx based on linux

    when: ansible_distribution == "Ubuntu"
    apt:
     name: nginx
     state: present

    when: ansible_distribution == "CentOS"
    yum: 
     name: nginx
     state: present
   
  - name: install python modules using pip
    pip:
     name: "{{ item }}"
     state: present
    with_items:
    - django
    - flask
    - request
    - os
